Program Menghitung Berat Badan Ideal (BBI)





def hitung_bbi(jenis_kelamin, tinggi): if jenis_kelamin.lower() == "laki-laki": bbi = (tinggi - 100) - (0.10 * (tinggi - 100)) elif jenis_kelamin.lower() == "perempuan": bbi = (tinggi - 100) - (0.15 * (tinggi - 100)) else: return None return bbi from js import document def hitung(event): jenis_kelamin = document.getElementById("jenis_kelamin").value tinggi = float(document.getElementById("tinggi").value) hasil = hitung_bbi(jenis_kelamin, tinggi) hasil_el = document.getElementById("hasil") if hasil is not None: hasil_el.innerText = f"Berat badan ideal untuk {jenis_kelamin} dengan tinggi {tinggi:.1f} cm adalah {hasil:.2f} kg" else: hasil_el.innerText = "Jenis kelamin tidak valid." document.getElementById("hitung").addEventListener("click", hitung)
