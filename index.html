Program Menghitung Berat Badan Ideal (BBI)





def hitung_bbi(jenis_kelamin, tinggi): if jenis_kelamin == "laki-laki": return (tinggi - 100) - 0.10 * (tinggi - 100) elif jenis_kelamin == "perempuan": return (tinggi - 100) - 0.15 * (tinggi - 100) else: return None from js import document def hitung(event): jk = document.getElementById("jenis_kelamin").value tb = float(document.getElementById("tinggi").value) hasil = hitung_bbi(jk, tb) out = document.getElementById("hasil") out.innerText = f"BBI untuk {jk} dengan tinggi {tb:.1f} cm adalah {hasil:.2f} kg" document.getElementById("hitung").addEventListener("click", hitung)
